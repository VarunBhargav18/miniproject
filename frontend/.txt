        document.getElementById('examStatus').addEventListener('change', function () {
            const examStatus = this.value;
            if (examStatus === 'yes') {
                document.getElementById('Yes').style.display = 'block';
                document.getElementById('No').style.display = 'none';
            } else {
                document.getElementById('Yes').style.display = 'none';
                document.getElementById('No').style.display = 'block';
            }
        });

        document.getElementById('examStatusForm').addEventListener('submit', async function (e) {
            e.preventDefault(); // Prevent the default form submission

            const examStatus = document.getElementById('examStatus').value;
            const rank = document.getElementById('rank') ? document.getElementById('rank').value : null; // Check if rank input exists
            const branch = document.getElementById('branch') ? document.getElementById('branch').value : null; // Check if branch input exists
            const category = document.getElementById('category').value; // Common category selection

            // Save user input in local storage (optional)
            localStorage.setItem('examStatus', examStatus);
            localStorage.setItem('rank', rank);
            localStorage.setItem('branch', branch);

            // Fetch college data based on user input
            try {
                const response = await fetch('/api/colleges'); // Replace with your actual API endpoint
                if (!response.ok) {
                    throw new Error('Network response was not ok');
                }
                const colleges = await response.json();
                
                // Filter or display colleges based on the user inputs (rank, branch, category)
                const collegeDataDiv = document.getElementById('college-data');
                collegeDataDiv.innerHTML = ''; // Clear previous data

                if (examStatus === 'yes' && rank) {
                    const filteredColleges = colleges.filter(college => {
                        // Assuming you have some logic to filter colleges by rank and category
                        return college[category] && college[category] <= rank; // Example filtering logic
                    });

                    if (filteredColleges.length === 0) {
                        collegeDataDiv.innerHTML = '<p>No colleges found for the given rank and category.</p>';
                    } else {
                        filteredColleges.forEach(college => {
                            const collegeDiv = document.createElement('div');
                            collegeDiv.innerHTML = `<h2>${college.name}</h2><p>${category}: ${college[category]}</p>`;
                            collegeDataDiv.appendChild(collegeDiv);
                        });
                    }
                } else if (examStatus === 'no' && branch) {
                    const filteredColleges = colleges.filter(college => college.branch === branch);
                    
                    if (filteredColleges.length === 0) {
                        collegeDataDiv.innerHTML = '<p>No colleges found for the selected branch.</p>';
                    } else {
                        filteredColleges.forEach(college => {
                            const collegeDiv = document.createElement('div');
                            collegeDiv.innerHTML = `<h2>${college.name}</h2>`;
                            collegeDataDiv.appendChild(collegeDiv);
                        });
                    }
                }
            } catch (error) {
                console.error('Error fetching colleges:', error);
                document.getElementById('college-data').innerHTML = '<p>Error fetching data.</p>';
            }
        });
    </script>
